1//! # Conversion Library
2//!
3//! A simple engineering unit conversion library for common measurements.
4//!
5//! This library provides functions for converting between different units of measurement
6//! commonly used in engineering projects: length, temperature, pressure, and mass.
7//!
8//! # Examples
9//!
10//! ```
11//! use conversion_lib::length;
12//! use conversion_lib::temperature;
13//!
14//! // Convert meters to feet
15//! let feet = length::meters_to_feet(100.0);
16//! assert_eq!(feet, 328.084);
17//!
18//! // Convert Celsius to Fahrenheit
19//! let fahrenheit = temperature::celsius_to_fahrenheit(25.0);
20//! assert_eq!(fahrenheit, 77.0);
21//! ```
22
23/// Length conversion functions
24pub mod length {
25    /// Converts meters to feet
26    ///
27    /// # Arguments
28    ///
29    /// * `meters` - Length in meters
30    ///
31    /// # Returns
32    ///
33    /// Length in feet
34    ///
35    /// # Examples
36    ///
37    /// ```
38    /// use conversion_lib::length;
39    ///
40    /// let feet = length::meters_to_feet(10.0);
41    /// assert_eq!(feet, 32.808399999999995);
42    /// ```
43    pub fn meters_to_feet(meters: f64) -> f64 {
44        meters * 3.28084
45    }
46
47    /// Converts feet to meters
48    ///
49    /// # Arguments
50    ///
51    /// * `feet` - Length in feet
52    ///
53    /// # Returns
54    ///
55    /// Length in meters
56    ///
57    /// # Examples
58    ///
59    /// ```
60    /// use conversion_lib::length;
61    ///
62    /// let meters = length::feet_to_meters(100.0);
63    /// assert_eq!(meters, 30.48);
64    /// ```
65    pub fn feet_to_meters(feet: f64) -> f64 {
66        feet * 0.3048
67    }
68
69    /// Converts kilometers to miles
70    ///
71    /// # Arguments
72    ///
73    /// * `km` - Distance in kilometers
74    ///
75    /// # Returns
76    ///
77    /// Distance in miles
78    pub fn km_to_miles(km: f64) -> f64 {
79        km * 0.621371
80    }
81
82    /// Converts miles to kilometers
83    pub fn miles_to_km(miles: f64) -> f64 {
84        miles * 1.60934
85    }
86}

88/// Temperature conversion functions
89pub mod temperature {
90    /// Converts Celsius to Fahrenheit
91    ///
92    /// # Arguments
93    ///
94    /// * `celsius` - Temperature in Celsius
95    ///
96    /// # Returns
97    ///
98    /// Temperature in Fahrenheit
99    ///
100    /// # Examples
101    ///
102    /// ```
103    /// use conversion_lib::temperature;
104    ///
105    /// let fahrenheit = temperature::celsius_to_fahrenheit(0.0);
106    /// assert_eq!(fahrenheit, 32.0);
107    /// ```
108    pub fn celsius_to_fahrenheit(celsius: f64) -> f64 {
109        (celsius * 9.0 / 5.0) + 32.0
110    }
111
112    /// Converts Fahrenheit to Celsius
113    pub fn fahrenheit_to_celsius(fahrenheit: f64) -> f64 {
114        (fahrenheit - 32.0) * 5.0 / 9.0
115    }
116
117    /// Converts Celsius to Kelvin
118    pub fn celsius_to_kelvin(celsius: f64) -> f64 {
119        celsius + 273.15
120    }
121
122    /// Converts Kelvin to Celsius
123    pub fn kelvin_to_celsius(kelvin: f64) -> f64 {
124        kelvin - 273.15
125    }
126}

128/// Pressure conversion functions
129pub mod pressure {
130    /// Converts bar to PSI (pounds per square inch)
131    ///
132    /// # Arguments
133    ///
134    /// * `bar` - Pressure in bar
135    ///
136    /// # Returns
137    ///
138    /// Pressure in PSI
139    pub fn bar_to_psi(bar: f64) -> f64 {
140        bar * 14.5038
141    }
142
143    /// Converts PSI to bar
144    pub fn psi_to_bar(psi: f64) -> f64 {
145        psi * 0.0689476
146    }
147
148    /// Converts Pascal to bar
149    pub fn pascal_to_bar(pascal: f64) -> f64 {
150        pascal / 100000.0
151    }
152
153    /// Converts bar to Pascal
154    pub fn bar_to_pascal(bar: f64) -> f64 {
155        bar * 100000.0
156    }
157}

159/// Mass conversion functions
160pub mod mass {
161    /// Converts kilograms to pounds
162    pub fn kg_to_pounds(kg: f64) -> f64 {
163        kg * 2.20462
164    }
165
166    /// Converts pounds to kilograms
167    pub fn pounds_to_kg(pounds: f64) -> f64 {
168        pounds * 0.453592
169    }
170
171    /// Converts metric tons (tonnes) to short tons
172    pub fn tonnes_to_tons(tonnes: f64) -> f64 {
173        tonnes * 1.10231
174    }
175
176    /// Converts short tons to metric tons (tonnes)
177    pub fn tons_to_tonnes(tons: f64) -> f64 {
178        tons * 0.907185
179    }
180}

182#[cfg(test)]
183mod tests {
184    use super::*;
185
186    #[test]
187    fn test_length_conversions() {
188        // Test meters to feet
189        let result = length::meters_to_feet(1.0);
190        assert!((result - 3.28084).abs() < 0.0001);
191
192        // Test round trip conversion
193        let meters = 100.0;
194        let feet = length::meters_to_feet(meters);
195        let back_to_meters = length::feet_to_meters(feet);
196        assert!((meters - back_to_meters).abs() < 0.0001);
197    }
198
199    #[test]
200    fn test_temperature_conversions() {
201        // Test freezing point
202        assert_eq!(temperature::celsius_to_fahrenheit(0.0), 32.0);
203        
204        // Test boiling point
205        assert_eq!(temperature::celsius_to_fahrenheit(100.0), 212.0);
206
207        // Test Kelvin
208        assert_eq!(temperature::celsius_to_kelvin(0.0), 273.15);
209    }
210
211    #[test]
212    fn test_pressure_conversions() {
213        let bar = 1.0;
214        let psi = pressure::bar_to_psi(bar);
215        let back_to_bar = pressure::psi_to_bar(psi);
216        assert!((bar - back_to_bar).abs() < 0.0001);
217    }
218
219    #[test]
220    fn test_mass_conversions() {
221        let kg = 100.0;
222        let pounds = mass::kg_to_pounds(kg);
223        let back_to_kg = mass::pounds_to_kg(pounds);
224        assert!((kg - back_to_kg).abs() < 0.0001);
225    }
